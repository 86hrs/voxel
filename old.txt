cmake_minimum_required(VERSION 3.5)
project(main)

# Set C++ standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_COMPILER "/usr/bin/clang++-19")

# Compilation flags
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -std=c++20")
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Define output directories
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/obj)

# Find required packages
set(OpenGL_GL_PREFERENCE GLVND)
find_package(OpenGL REQUIRED)
include_directories(${OPENGL_INCLUDE_DIRS}
imgui
imgui/backends
)

set(IMGUI_DIR ${CMAKE_SOURCE_DIR}/imgui)
set(IMGUI_SOURCES
    ${IMGUI_DIR}/imgui.cpp
    ${IMGUI_DIR}/imgui_demo.cpp
    ${IMGUI_DIR}/imgui_draw.cpp
    ${IMGUI_DIR}/imgui_tables.cpp
    ${IMGUI_DIR}/imgui_widgets.cpp
    ${IMGUI_DIR}/backends/imgui_impl_glfw.cpp
    ${IMGUI_DIR}/backends/imgui_impl_opengl3.cpp
)

# Add GLFW as a subdirectory
set(GLFW_BUILD_DOCS OFF CACHE BOOL "GLFW lib only")
set(GLFW_INSTALL OFF CACHE BOOL "GLFW lib only")
add_subdirectory(glfw)

file(GLOB_RECURSE SRC_FILES CONFIGURE_DEPENDS src/*.cpp src/*.c src/*.h src/*.cc)

# Create executable
add_executable(${PROJECT_NAME} ${SRC_FILES} ${IMGUI_SOURCES})

# Link libraries
target_link_libraries(${PROJECT_NAME}
    ${OPENGL_LIBRARIES}
    glfw
    freetype
)

# Include src directory for headers
target_include_directories(${PROJECT_NAME} PRIVATE 
    src
    ${IMGUI_DIR}
    ${IMGUI_DIR}/backends
    ${FREETYPE_INCLUDE_DIRS}
)

file(GLOB BLOCK_TEXTURES "${CMAKE_SOURCE_DIR}/src/block/*.png")
# List of files to copy to build folder
set(SHADER_FILES
    ${CMAKE_SOURCE_DIR}/src/vertex.glsl
    ${CMAKE_SOURCE_DIR}/src/fragment.glsl
    ${CMAKE_SOURCE_DIR}/src/hud_fragment.glsl
    ${CMAKE_SOURCE_DIR}/src/hud_vertex.glsl
    ${BLOCK_TEXTURES}
)

# Define a custom target to copy only the specified shader files
add_custom_target(copy_shaders ALL
    COMMENT "Copying shaders to output directory"
)

# Add individual file copy commands for each shader file
foreach(SHADER_FILE ${SHADER_FILES})
    get_filename_component(SHADER_FILENAME ${SHADER_FILE} NAME)
    add_custom_command(
        TARGET copy_shaders
        COMMAND ${CMAKE_COMMAND} -E copy
        ${SHADER_FILE}
        ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/${SHADER_FILENAME}
    )
endforeach()

# Make sure shaders are copied before the main executable is built
add_dependencies(${PROJECT_NAME} copy_shaders)
